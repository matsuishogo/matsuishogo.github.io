N,K = map(int,input().split())
D = list(map(int,input().split()))
use = [x for x in range(10)]
use = list(set(use) ^ set(D))
use.sort()
N = str(N)
ans = ""
count = 0
for i in range(len(N)):
if count == 0:  
    for j in use:
    if j >= int(N[i]):
        ans += str(j)
        break
    if ans == "":
    ans += str(use[0]) if use[0]!=0 else str(use[1])
    ans += str(use[0])*len(N)

    break
else:
    for j in use:
    if j >= int(N[i]):
        ans += str(j)
        break
count += 1
print(ans)
